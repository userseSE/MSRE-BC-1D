cmake_minimum_required(VERSION 3.10)

# Project name
project(MSR1DCPP)

# Set the C++ standard to C++17
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# Include directories for Eigen and matplotlibcpp
include_directories(${CMAKE_SOURCE_DIR}/thirdparty/eigen)
include_directories(${CMAKE_SOURCE_DIR}/thirdparty/matplotlibcpp)

# Manually specify the Python executable and include directories
set(Python3_EXECUTABLE "/Users/ouuyou/miniforge/bin/python3")
include_directories("/Users/ouuyou/miniforge/include/python3.10")
include_directories("/Users/ouuyou/miniforge/lib/python3.10/site-packages/numpy/core/include")

# Add include directory to the compiler's search path
include_directories("${CMAKE_SOURCE_DIR}/include")

# Collect all source files in the src directory
file(GLOB SOURCES "src/*.cpp")

# Add the executable
add_executable(test ${SOURCES})

# Link against Python libraries
target_link_libraries(test ${Python3_LIBRARIES} "/Users/ouuyou/miniforge/lib/libpython3.10.dylib")
